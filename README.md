## Install

```bash
npm install --save celik-ui
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'celik-ui'
import 'celik-ui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
